🟠 HTML5 - Structure de page

La structure HTML5 complète d'une page sémantique

<!-- Déclaration DOCTYPE : obligatoire en HTML5 -->
<!DOCTYPE html>

<!-- lang="fr" : essentiel pour l'accessibilité -->
<html lang="fr">

<head>
  <!-- UTF-8 : encode tous les caractères spéciaux -->
  <meta charset="UTF-8"/>

  <!-- viewport : INDISPENSABLE pour le responsive -->
  <meta name="viewport"
        content="width=device-width, initial-scale=1.0"/>

  <!-- description : utilisée par les moteurs de recherche -->
  <meta name="description"
        content="Description concise de la page (150-160 car.)"/>

  <!-- title : affiché dans l'onglet et résultats Google -->
  <title>Titre de la page - Nom du site</title>

  <!-- Feuille de style externe -->
  <link rel="stylesheet" href="css/style.css"/>
</head>

<body>

  <!-- header : en-tête global de la page -->
  <header role="banner">
    <nav aria-label="Navigation principale">
      <!-- menu ici -->
    </nav>
  </header>

  <!-- main : contenu principal, UNIQUE par page -->
  <main id="contenu-principal">

    <!-- section : thème distinct, doit avoir un titre -->
    <section aria-labelledby="titre-section">
      <h2 id="titre-section">Titre de section</h2>

      <!-- article : contenu autonome (card, post...) -->
      <article>
        <h3>Titre de l'article</h3>
        <p>Contenu...</p>
      </article>
    </section>

  </main>

  <!-- footer : pied de page, mentions légales -->
  <footer role="contentinfo">
    <p>&copy; <time datetime="2025">2025</time> Mon Site</p>
  </footer>

  <!-- JS en bas du body = ne bloque pas le rendu -->
  <script src="js/app.js"></script>

</body>
</html>
DOCTYPEActive le mode HTML5 standard dans tous les navigateurs
lang="fr"Lecteurs d'écran, SEO, correcteurs orthographiques en dépendent
viewportSans cette meta, le site s'affiche comme une version desktop sur mobile
role="banner"ARIA pour header global, redondant mais recommandé
aria-label navDistingue la nav principale des autres nav de la page
aria-labelledbyLie la section à son titre - meilleure pratique d'accessibilité
script en basGarantit que le DOM est chargé avant d'exécuter le JS

🟠 HTML5 - Formulaire accessible

Un formulaire de contact avec toutes les bonnes pratiques

HTMLformulaire-contact.html
<!-- fieldset groupe les champs liés, legend les décrit -->
<form id="form-contact"
      novalidate
      aria-label="Formulaire de contact">

  <fieldset>
    <legend>Vos informations</legend>

    <!-- Groupe : label + input + message d'erreur -->
    <div class="field-group">

      <!-- for="id" associe le label à l'input -->
      <label for="nom">
        Nom <span aria-hidden="true">*</span>
      </label>

      <input
        type="text"
        id="nom"
        name="nom"
        required
        autocomplete="family-name"
        aria-describedby="nom-erreur"
        aria-required="true"
        placeholder="Dupont"
      />

      <!-- role="alert" annonce l'erreur aux lecteurs d'écran -->
      <span
        id="nom-erreur"
        class="field-error"
        role="alert"
        aria-live="polite"
      ></span>

    </div>

    <!-- Champ email avec validation HTML5 native -->
    <div class="field-group">
      <label for="email">Email</span aria-hidden="true">*</span></label>
      <input type="email" id="email" name="email"
             required autocomplete="email"
             aria-describedby="email-erreur"/>
      <span id="email-erreur" class="field-error" role="alert"></span>
    </div>

  </fieldset>

  <!-- type="submit" : déclenche l'événement submit -->
  <button type="submit">Envoyer le message</button>

</form>
novalidate sur le form désactive la validation HTML5 native pour gérer soi-même les erreurs en JS (meilleur contrôle du design et des messages). aria-describedby pointe vers le <span> d'erreur - quand le JS y injecte un message, le lecteur d'écran l'annonce automatiquement.

🔵 CSS3 - Media queries Responsive

Approche mobile-first avec des breakpoints progressifs

✓ Mobile-first (recommandé)
/* Base : mobile (default) */
.grid {
  display: grid;
  /* 1 colonne sur mobile */
  grid-template-columns: 1fr;
  gap: 1rem;
}

/* Tablette : ≥ 600px */
@media (min-width: 600px) {
  .grid {
    grid-template-columns:
      repeat(2, 1fr);
  }
}

/* Desktop : ≥ 900px */
@media (min-width: 900px) {
  .grid {
    grid-template-columns:
      repeat(3, 1fr);
  }
}
✗ Desktop-first (à éviter)
/* Base : desktop */
.grid {
  display: grid;
  /* 3 colonnes de base */
  grid-template-columns:
    repeat(3, 1fr);
}

/* Tablette */
@media (max-width: 900px) {
  .grid {
    grid-template-columns:
      repeat(2, 1fr);
  }
}

/* Mobile */
@media (max-width: 600px) {
  .grid {
    grid-template-columns:
      1fr;
  }
}

🟡 JavaScript - fetch() AJAX complet

Chargement asynchrone de données JSON avec gestion d'erreur

JSfetch-formations.js
/**
 * Charge et affiche les formations depuis un fichier JSON.
 * Utilise fetch() avec async/await - approche moderne ES2017+
 */

// ── Sélection des éléments DOM ─────────────────────
const grille  = document.getElementById('formations-grid');
const loader  = document.querySelector('.loader');

// ── Fonction principale asynchrone ──────────────────
async function chargerFormations() {

  // 1. Afficher le loader pendant le chargement
  afficherLoader(true);

  // try/catch : gère les erreurs réseau et JSON
  try {

    // 2. Envoyer la requête HTTP GET
    const response = await fetch('./data/formations.json');

    // 3. Vérifier que la réponse est un succès (200-299)
    if (!response.ok) {
      // Lance une erreur qui sera attrapée par le catch
      throw new Error(`Erreur HTTP : ${response.status}`);
    }

    // 4. Parser la réponse en JSON (aussi async !)
    //    data = { formations: [...] }
    const data = await response.json();

    // 5. Extraire le tableau (avec valeur par défaut [])
    const formations = data.formations ?? [];

    // 6. Afficher les formations dans le DOM
    afficherFormations(formations);

  } catch (erreur) {

    // 7. Afficher un message d'erreur utilisateur
    console.error('Erreur de chargement :', erreur);
    afficherErreur('Impossible de charger les formations.');

  } finally {

    // 8. finally : toujours exécuté (succès ou erreur)
    afficherLoader(false);

  }
}

// ── Fonctions utilitaires ────────────────────────────

/**
 * Génère et insère les cartes de formation.
 * @param {Array} formations - Tableau d'objets formation
 */
function afficherFormations(formations) {
  // Vider la grille avant d'insérer les nouvelles cartes
  grille.innerHTML = '';

  if (formations.length === 0) {
    grille.innerHTML = '<p>Aucune formation disponible.</p>';
    return;
  }

  // forEach : parcourt chaque formation
  formations.forEach((formation, index) => {
    const carte = creerCarte(formation, index);
    grille.appendChild(carte);   // Ajoute au DOM
  });
}

/**
 * Crée un élément article pour une formation.
 * Utilise createElement (plus sûr que innerHTML pour les données externes)
 */
function creerCarte(formation, index) {
  const article = document.createElement('article');
  article.className = 'formation-card';

  // animation-delay : chaque carte apparaît avec un léger décalage
  article.style.animationDelay = `${index * 60}ms`;

  // textContent est sûr (pas d'injection XSS possible)
  article.innerHTML = `
    <div class="card-header">
      <span class="card-level">${echapper(formation.level)}</span>
    </div>
    <h3 class="card-title">${echapper(formation.title)}</h3>
    <p class="card-category">${echapper(formation.category)}</p>
    <p class="card-duration">Durée : ${echapper(formation.duration)}</p>
  `;

  return article;
}

/**
 * Échappe les caractères HTML - prévention XSS.
 * Toujours échapper les données avant insertion dans le DOM.
 */
function echapper(chaine) {
  const div = document.createElement('div');
  div.textContent = String(chaine);
  return div.innerHTML;
}

// ── Lancement au chargement du DOM ──────────────────
// DOMContentLoaded : le DOM est prêt, pas besoin d'attendre les images
document.addEventListener('DOMContentLoaded', chargerFormations);

🟡 JavaScript - Validation de formulaire

Validation complète avec messages d'erreur accessibles

JSform-validation.js
/**
 * Validation de formulaire en JavaScript pur.
 * Principes : accessibilité, messages clairs, aria-invalid.
 */

const form     = document.querySelector('#form-contact');
const champNom  = document.getElementById('nom');
const champEmail= document.getElementById('email');

// ── Règles de validation ─────────────────────────────
const regles = {
  nom: {
    // Valider retourne null si OK, ou un message d'erreur
    valider: (val) => {
      if (!val.trim()) return 'Le nom est obligatoire.';
      if (val.length < 2)  return 'Le nom doit contenir au moins 2 caractères.';
      return null;  // null = pas d'erreur
    }
  },
  email: {
    valider: (val) => {
      if (!val.trim()) return 'L\'email est obligatoire.';
      // Regex simple pour valider le format email
      if (!/^[^\s@]+@[^\s@]+\.[^\s@]+$/.test(val))
        return 'L\'adresse email n\'est pas valide.';
      return null;
    }
  }
};

// ── Afficher / effacer une erreur ────────────────────
function afficherErreurChamp(input, messageErreur) {
  // Trouver le span d'erreur associé (via aria-describedby)
  const spanErreur = document.getElementById(
    input.getAttribute('aria-describedby')
  );

  if (messageErreur) {
    // ERREUR : signaler visuellement ET aux technologies d'assistance
    input.setAttribute('aria-invalid', 'true');   // lecteur d'écran
    input.classList.add('input-error');           // style CSS rouge
    if (spanErreur) spanErreur.textContent = messageErreur;
  } else {
    // SUCCÈS : réinitialiser tout
    input.setAttribute('aria-invalid', 'false');
    input.classList.remove('input-error');
    input.classList.add('input-success');
    if (spanErreur) spanErreur.textContent = '';
  }
}

// ── Validation en temps réel (blur = quand on quitte le champ) ──
[champNom, champEmail].forEach(input => {
  input.addEventListener('blur', () => {
    const regle = regles[input.name];
    if (!regle) return;
    const erreur = regle.valider(input.value);
    afficherErreurChamp(input, erreur);
  });
});

// ── Soumission du formulaire ─────────────────────────
form?.addEventListener('submit', (event) => {

  // Empêche le rechargement de page (comportement natif)
  event.preventDefault();

  // Valider tous les champs
  const erreurs = {
    nom:   regles.nom.valider(champNom.value),
    email: regles.email.valider(champEmail.value),
  };

  // Afficher les erreurs
  afficherErreurChamp(champNom,   erreurs.nom);
  afficherErreurChamp(champEmail, erreurs.email);

  // S'il n'y a aucune erreur : soumettre
  const formulaireValide = !erreurs.nom && !erreurs.email;

  if (formulaireValide) {
    envoyerFormulaire(form);
  } else {
    // Focus sur le premier champ en erreur
    document.querySelector('.input-error')?.focus();
  }
});

🟡 JavaScript - Menu burger accessible

Implémentation complète avec accessibilité clavier

JSmenu-burger.js
/**
 * Menu burger mobile - Version accessible complète.
 * Gère : clic bouton, clic extérieur, touche Escape, scroll body.
 */

const burger      = document.getElementById('burger-btn');
const menuMobile  = document.getElementById('mobile-menu');

// ── Ouvrir / fermer (toggle) ─────────────────────────
function toggleMenu() {
  const estOuvert = menuMobile.classList.toggle('open');

  // Synchroniser l'attribut ARIA pour les lecteurs d'écran
  burger.setAttribute('aria-expanded', String(estOuvert));

  // Bloquer le scroll du body quand le menu est ouvert
  document.body.style.overflow = estOuvert ? 'hidden' : '';
}

// ── Fermer uniquement ────────────────────────────────
function fermerMenu() {
  menuMobile.classList.remove('open');
  burger.setAttribute('aria-expanded', 'false');
  document.body.style.overflow = '';
}

// ── Événements ──────────────────────────────────────

// 1. Clic sur le bouton burger
burger?.addEventListener('click', toggleMenu);

// 2. Clic sur un lien du menu → fermer
menuMobile?.querySelectorAll('a').forEach(lien => {
  lien.addEventListener('click', fermerMenu);
});

// 3. Clic en dehors du menu → fermer
//    Vérifie que le clic n'est ni dans le menu ni sur le burger
document.addEventListener('click', (e) => {
  if (
    menuMobile?.classList.contains('open') &&
    !menuMobile.contains(e.target) &&       // pas dans le menu
    !burger.contains(e.target)              // pas sur le burger
  ) {
    fermerMenu();
  }
});

// 4. Touche Escape → fermer + remettre le focus sur le burger
document.addEventListener('keydown', (e) => {
  if (e.key === 'Escape' && menuMobile?.classList.contains('open')) {
    fermerMenu();
    burger?.focus();  // Ramène le focus au bouton burger
  }
});
Ce pattern est complet et accessible : l'attribut aria-expanded est synchronisé à chaque état, le focus est géré au clavier, et le scroll du body est bloqué pour éviter que l'utilisateur scroll en arrière-plan pendant que le menu est ouvert.
